    Script logic:
    ```
    const BASE = import.meta.env.BASE_URL;
    let pagefind: any;

    async function loadPagefind() {
      if (!pagefind) {
        pagefind = await import(/* @vite-ignore */ `${BASE}pagefind/pagefind.js`);
        await pagefind.init();
      }
      return pagefind;
    }

    // Get both desktop and mobile elements
    const desktopInput = document.getElementById('search-input') as HTMLInputElement;
    const mobileInput = document.getElementById('mobile-search-input') as HTMLInputElement;
    const desktopDropdown = document.getElementById('search-dropdown') as HTMLElement;
    const mobileDropdown = document.getElementById('mobile-search-dropdown') as HTMLElement;
    const searchContainer = document.getElementById('search-container');
    const mobileSearchContainer = document.getElementById('mobile-search');
    const searchToggle = document.getElementById('search-toggle');
    const mobileSearchDiv = document.getElementById('mobile-search');

    // Mobile search toggle
    searchToggle?.addEventListener('click', () => {
      mobileSearchDiv?.classList.toggle('hidden');
      if (!mobileSearchDiv?.classList.contains('hidden')) {
        mobileInput?.focus();
      }
    });

    // Shared search handler
    function setupSearch(input: HTMLInputElement, dropdown: HTMLElement) {
      let debounce: ReturnType<typeof setTimeout>;
      input.addEventListener('input', () => {
        const q = input.value.trim();
        clearTimeout(debounce);
        if (q.length < 3) {
          dropdown.innerHTML = '';
          dropdown.classList.add('hidden');
          return;
        }
        debounce = setTimeout(async () => {
          const pf = await loadPagefind();
          const result = await pf.debouncedSearch(q);
          if (!result) return;
          const top5 = await Promise.all(result.results.slice(0, 5).map((r: any) => r.data()));
          renderDropdown(top5, dropdown);
        }, 200);
      });
    }

    function renderDropdown(results: any[], dropdown: HTMLElement) {
      if (results.length === 0) {
        dropdown.innerHTML = '<p class="p-3 text-oni-text-muted text-sm">No results found</p>';
      } else {
        dropdown.innerHTML = results.map(r => `
          <a href="${r.url}" class="flex items-center justify-between gap-3 px-3 py-2 hover:bg-oni-bg-elevated transition-colors">
            <span class="text-oni-text-primary text-sm truncate">${r.meta.title}</span>
            ${r.meta.category ? `<span class="text-xs px-2 py-0.5 rounded bg-oni-teal/10 text-oni-teal whitespace-nowrap">${r.meta.category}</span>` : ''}
          </a>
        `).join('');
      }
      dropdown.classList.remove('hidden');
    }

    if (desktopInput && desktopDropdown) setupSearch(desktopInput, desktopDropdown);
    if (mobileInput && mobileDropdown) setupSearch(mobileInput, mobileDropdown);

    // Close on Escape (close all dropdowns, keep query text)
    document.addEventListener('keydown', (e) => {
      if (e.key === 'Escape') {
        desktopDropdown?.classList.add('hidden');
        mobileDropdown?.classList.add('hidden');
      }
    });

    // Close on click outside
    document.addEventListener('click', (e) => {
      const target = e.target as Node;
      if (searchContainer && !searchContainer.contains(target)) {
        desktopDropdown?.classList.add('hidden');
      }
      if (mobileSearchContainer && !mobileSearchContainer.contains(target) && !(searchToggle?.contains(target))) {
        mobileDropdown?.classList.add('hidden');
      }
    });
    ```

    - Use a uniform teal badge color for all categories (bg-oni-teal/10 text-oni-teal) per discretion
    - No loading state needed -- pagefind loads fast enough on a static site
    - Dropdown items are `<a>` tags so clicking navigates to the result page
